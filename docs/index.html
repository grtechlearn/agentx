Quick Start

Up and Running in 30 Seconds

From zero to production-ready agent with just a few lines of code.

from agentx import Agent, Config

# Create an agent with one line
config = Config(
    llm_provider="openai",
    model="gpt-4",
    temperature=0.7
)

agent = Agent(
    name="researcher",
    role="Research Assistant",
    config=config
)

# Run with a single call
result = await agent.run("Analyze Q4 market trends")
print(result.output)
from agentx import Daemon, Agent, Schedule

# Create a 24/7 autonomous agent
daemon = Daemon(
    agent=Agent(name="monitor", role="System Monitor"),
    schedule=Schedule.every(30).minutes,
    health_check=True,
    auto_restart=True
)

# Start the daemon
daemon.start()

# It runs in the background, checking every 30 min
# with automatic health checks and restart on failure
from agentx import Agent, StreamConfig

agent = Agent(
    name="writer",
    role="Content Writer",
    stream=StreamConfig(mode="sse")
)

# Stream tokens in real-time
async for chunk in agent.stream("Write a blog post about AI agents"):
    print(chunk.token, end="", flush=True)

# Or use WebSocket for bidirectional streaming
async with agent.ws_stream() as ws:
    await ws.send("Start writing")
    async for msg in ws:
        print(msg.data)
